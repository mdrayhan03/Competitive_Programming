int n , magnet ;
int prev_magnet , cnt = 0 ;

int main()
{
    cin >> n ;
    for (int i = 0 ; i < n ; i++) {
        cin >> magnet ;
        if (prev_magnet != magnet) {
            cnt ++ ;
        }
        prev_magnet = magnet ;
    }

def function (n) :
    cnt = 0
    if (n % 2 != 0) :
        cnt = (n // 2) - n
    
    else :
        cnt = (n // 2)
    
    return cnt
